def plot_colored_square(img, ax = None, metric = 'avg'):
    if ax is None:
        fig = plt.figure()
        ax = fig.add_subplot(111, aspect = 'equal')
        
    if metric not in ['avg', 'max']:
        assert('Not a valid metric')
        
    colors = np.array([sum(img[:,:,i].flatten()) for i in range(0,3)]) / len(img[:,:,0].flatten())
    
    if metric == 'avg':
        ax.add_patch(
        patches.Rectangle((0.0, 0.0), 1.0, 1.0,  
                      color = colors)
        )
        
    if metric == 'max':
        ax.add_patch(
        patches.Rectangle((0.0, 0.0), 1.0, 1.0,  
                      color = [int(x) for x in colors / max(colors)])
        )
    return colors

for i, row in enumerate(axes):
    for j, ax in enumerate(row):
        if (i,j) in loc_states_custom.keys():
            state = loc_states_custom[(i,j)]
            flag = mpimg.imread('flags-normal/' + us_state_abbrev[state].lower() + '.png')
            
            colors = plot_colored_square(flag, ax = ax, metric='avg')
            
            if sum(colors) > 1.5:
                text = 'black'
            else:
                text = 'white'
                
            ax.text(.5,.5,us_state_abbrev[state].upper(),
                    horizontalalignment='center',
                    transform=ax.transAxes, color=text, size=20)
                
            ax.get_xaxis().set_ticks([])
            ax.get_yaxis().set_ticks([]) 
        else:
            ax.axis('off')     
            
fig.suptitle('Average State Map Color', size = 38)
plt.show()
